"""
db.py — Async SQLAlchemy engine + session factory + claim helper.

Two connection modes:

  * Cloud Run: Cloud SQL Python Connector with pg8000 driver + IAM auth.
               Same pattern as the rest of our GCP services. Triggered by
               INSTANCE_CONNECTION_NAME being set.
  * Local:    Plain pg8000 DSN built from DB_HOST/DB_PORT/etc.

We use pg8000 (pure-Python Postgres driver) because the Cloud SQL Python
Connector supports it natively with IAM auth via `enable_iam_auth=True`.
pg8000 is synchronous, so we wrap it with SQLAlchemy's standard sync engine
inside async runners. SQLAlchemy handles thread-pool offloading transparently
via `AsyncEngine` over a sync driver — no asyncpg needed.

The claim helper uses `SELECT ... FOR UPDATE SKIP LOCKED` so two sweep
instances can safely run concurrently without ever grabbing the same row.
"""
from __future__ import annotations

import os
from contextlib import asynccontextmanager
from datetime import datetime, timezone
from typing import AsyncIterator, Optional

import sqlalchemy
from sqlalchemy import select, text, update
from sqlalchemy.ext.asyncio import (
    AsyncEngine,
    AsyncSession,
    async_sessionmaker,
    create_async_engine,
)

import config
from logger import get_logger
from models import CLAIMABLE_STATUSES, Complaint, ComplaintStatus

log = get_logger(__name__)


# ---------------------------------------------------------------------------
# Connection pool builders
# ---------------------------------------------------------------------------

def _create_conn_pool_cloud_sql() -> AsyncEngine:
    """Build an async engine that uses the Cloud SQL Python Connector + pg8000.

    Mirrors the team-standard `_create_conn_pool` pattern:
      - pg8000 driver
      - enable_iam_auth=True (no password needed; ADC handles it)
      - PRIVATE/PUBLIC IP toggle via env
    """
    from google.cloud.sql.connector import Connector, IPTypes  # type: ignore[import-not-found]

    instance_connection_name = config.INSTANCE_CONNECTION_NAME
    log.info("instance connection name: %s", instance_connection_name)

    ip_type = (
        IPTypes.PRIVATE
        if os.getenv("PRIVATE_IP", "false").lower() == "true"
        else IPTypes.PUBLIC
    )

    connector = Connector()

    def getconn():
        conn = connector.connect(
            instance_connection_name,
            "pg8000",
            user=config.DB_USER,
            db=config.DB_NAME,
            enable_iam_auth=True,
            ip_type=ip_type,
        )
        return conn

    # The Cloud SQL Python Connector is used with SQLAlchemy via the
    # `creator` argument. We wrap with create_async_engine so the rest of
    # the app can keep using `async with session_scope()`. SQLAlchemy
    # automatically offloads sync pg8000 calls to a thread pool.
    engine = create_async_engine(
        "postgresql+pg8000://",
        creator=getconn,
        # Pool size — maximum number of permanent connections.
        pool_size=config.DB_POOL_SIZE,
        # Temporarily exceeds the set pool size if no connections are available.
        max_overflow=config.DB_MAX_OVERFLOW,
        # Pool timeout in seconds.
        pool_timeout=30,
        # Maximum number of seconds a connection persists; reestablished after.
        pool_recycle=1800,
        pool_pre_ping=True,
    )
    log.info(
        "DB engine ready (Cloud SQL via pg8000 + IAM) | ip_type=%s pool=%d+%d",
        ip_type.name, config.DB_POOL_SIZE, config.DB_MAX_OVERFLOW,
    )
    return engine


def _create_conn_pool_local() -> AsyncEngine:
    """Build an engine for local dev — plain pg8000 DSN with password."""
    dsn = (
        f"postgresql+pg8000://{config.DB_USER}:{config.DB_PASSWORD}"
        f"@{config.DB_HOST}:{config.DB_PORT}/{config.DB_NAME}"
    )
    engine = create_async_engine(
        dsn,
        pool_size=config.DB_POOL_SIZE,
        max_overflow=config.DB_MAX_OVERFLOW,
        pool_timeout=30,
        pool_recycle=1800,
        pool_pre_ping=True,
    )
    log.info(
        "DB engine ready (local pg8000) | host=%s:%d db=%s pool=%d+%d",
        config.DB_HOST, config.DB_PORT, config.DB_NAME,
        config.DB_POOL_SIZE, config.DB_MAX_OVERFLOW,
    )
    return engine


# ---------------------------------------------------------------------------
# Lifecycle
# ---------------------------------------------------------------------------

_engine: Optional[AsyncEngine] = None
_sessionmaker: Optional[async_sessionmaker[AsyncSession]] = None


async def init_db() -> None:
    """Build engine + sessionmaker once at startup."""
    global _engine, _sessionmaker
    if _engine is not None:
        return

    if config.INSTANCE_CONNECTION_NAME:
        _engine = _create_conn_pool_cloud_sql()
    else:
        _engine = _create_conn_pool_local()

    _sessionmaker = async_sessionmaker(_engine, expire_on_commit=False)

    # Sanity check
    async with _engine.connect() as conn:
        await conn.execute(text("SELECT 1"))
    log.info("DB connectivity verified")


async def close_db() -> None:
    global _engine, _sessionmaker
    if _engine is not None:
        await _engine.dispose()
        _engine = None
        _sessionmaker = None
        log.info("DB engine disposed")


def get_sessionmaker() -> async_sessionmaker[AsyncSession]:
    if _sessionmaker is None:
        raise RuntimeError("DB not initialised — call init_db() first")
    return _sessionmaker


@asynccontextmanager
async def session_scope() -> AsyncIterator[AsyncSession]:
    """Open a session that commits on success and rolls back on error."""
    sm = get_sessionmaker()
    async with sm() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise


# ---------------------------------------------------------------------------
# Claim helper — the heart of safe concurrent sweeping
# ---------------------------------------------------------------------------

async def claim_pending_complaints(
    session: AsyncSession,
    *,
    batch_size: int,
    max_attempts: int,
) -> list[int]:
    """Atomically claim up to `batch_size` complaints.

    Uses SELECT ... FOR UPDATE SKIP LOCKED so two sweeps never grab the same
    row. Each claimed row is bumped to PROCESSING + locked_at = now() before
    the transaction commits, so any concurrent reader sees the lock.

    Returns the list of complaint IDs claimed.
    """
    now = datetime.now(timezone.utc)

    stmt = (
        select(Complaint.id)
        .where(Complaint.status.in_(CLAIMABLE_STATUSES))
        .where(Complaint.attempt_count < max_attempts)
        .order_by(Complaint.created_at.asc())
        .limit(batch_size)
        .with_for_update(skip_locked=True)
    )
    result = await session.execute(stmt)
    ids = [row[0] for row in result.all()]

    if not ids:
        return []

    await session.execute(
        update(Complaint)
        .where(Complaint.id.in_(ids))
        .values(
            status=ComplaintStatus.PROCESSING,
            locked_at=now,
            updated_at=now,
            attempt_count=Complaint.attempt_count + 1,
        )
    )

    log.info("Claimed %d complaint(s): %s", len(ids), ids)
    return ids
