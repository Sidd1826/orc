"""
db.py — Postgres connection via pg8000 + SQLAlchemy, using plain
host / user / password / dbname (the same way you connect in DBeaver).

Two ways to point it at a database, in priority order:

  1. DATABASE_URL  — a full SQLAlchemy URL, e.g.
         postgresql+pg8000://user:pass@host:5432/dbname
     (handy for quick local overrides)

  2. PG_HOST / PG_PORT / PG_USER / PG_PASSWORD / PG_DB
     - LOCAL / TCP:   PG_HOST=10.20.30.40  (or localhost)
     - Cloud Run:     PG_HOST=/cloudsql/PROJECT:REGION:INSTANCE
                      (unix socket created by --add-cloudsql-instances; still
                       password auth — no IAM connector needed)

The Engine + sessionmaker are created lazily on first use.
"""
from __future__ import annotations

import os

import pg8000.dbapi
import sqlalchemy
from sqlalchemy.orm import Session, sessionmaker

import config
from logger import get_logger

log = get_logger(__name__)

_engine: sqlalchemy.engine.base.Engine | None = None
_SessionLocal: sessionmaker[Session] | None = None


def _create_conn_pool() -> sqlalchemy.engine.base.Engine:
    """Build a SQLAlchemy Engine over a pg8000 connection pool."""

    database_url = os.getenv("DATABASE_URL")
    if database_url:
        log.info("DB: using DATABASE_URL")
        return sqlalchemy.create_engine(
            database_url,
            pool_size=5, max_overflow=2, pool_timeout=30,
            pool_recycle=1800, pool_pre_ping=True,
        )

    host = config.PG_HOST
    port = config.PG_PORT
    user = config.PG_USER
    password = config.PG_PASSWORD
    db = config.PG_DB

    is_unix_socket = host.startswith("/")
    log.info("DB: host=%s port=%s db=%s user=%s mode=%s",
             host, port, db, user, "unix_socket" if is_unix_socket else "tcp")

    def getconn() -> pg8000.dbapi.Connection:
        if is_unix_socket:
            # Cloud Run mounts the socket dir; the socket file is .s.PGSQL.5432
            sock = host if host.endswith(".s.PGSQL.5432") else f"{host}/.s.PGSQL.5432"
            return pg8000.dbapi.connect(
                user=user, password=password, database=db, unix_sock=sock,
            )
        return pg8000.dbapi.connect(
            user=user, password=password, database=db, host=host, port=int(port),
        )

    return sqlalchemy.create_engine(
        "postgresql+pg8000://",
        creator=getconn,
        pool_size=5,            # permanent connections
        max_overflow=2,         # temporary extra connections under load
        pool_timeout=30,        # seconds to wait for a free connection
        pool_recycle=1800,      # recycle connections older than 30 min
        pool_pre_ping=True,     # validate before use (survives Cloud Run idle)
    )


def get_engine() -> sqlalchemy.engine.base.Engine:
    global _engine
    if _engine is None:
        _engine = _create_conn_pool()
    return _engine


def get_sessionmaker() -> sessionmaker[Session]:
    global _SessionLocal
    if _SessionLocal is None:
        _SessionLocal = sessionmaker(
            bind=get_engine(), autoflush=False, expire_on_commit=False
        )
    return _SessionLocal


def session_scope() -> Session:
    """Return a new Session. Use as a context manager:
        with session_scope() as s: ...
    """
    return get_sessionmaker()()


def ping() -> bool:
    """Quick connectivity check used by /health."""
    with get_engine().connect() as conn:
        conn.execute(sqlalchemy.text("SELECT 1"))
    return True


def dispose() -> None:
    global _engine
    if _engine is not None:
        _engine.dispose()
        _engine = None
