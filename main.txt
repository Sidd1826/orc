"""
main.py — FastAPI app for the fraud-ops orchestrator.

Routes
  GET  /health
  POST /api/v1/complaints           ingest one MHA complaint (writes RECEIVED)
  POST /process-batch               polling tick (called by Cloud Scheduler)
  POST /process/{ack_no}            manual single-complaint trigger (ops/debug)
  GET  /api/v1/complaints/{ack_no}  status + final response lookup

Run locally:
  uvicorn main:app --host 0.0.0.0 --port 8080 --reload
"""
from __future__ import annotations

import asyncio
import os
import uuid
from contextlib import asynccontextmanager

from fastapi import Body, FastAPI, HTTPException
from sqlalchemy import select

import config
import processor
from db import dispose, ping, session_scope
from logger import get_logger
from models import (
    ApiResponseLog, Complaint, ComplaintProcessing, Incident, Instrument,
    STATUS_OPEN,
)
from schemas import ComplaintIn, IngestAck

log = get_logger(__name__)


@asynccontextmanager
async def lifespan(app: FastAPI):
    log.info("=== Starting Fraud-Ops Orchestrator ===")
    log.info("gateway=%s threshold=%dh batch=%d",
             config.GATEWAY_BASE_URL, config.FRAUD_AGE_THRESHOLD_HOURS, config.BATCH_SIZE)
    try:
        yield
    finally:
        processor.gateway().close()
        dispose()
        log.info("=== Orchestrator shut down ===")


app = FastAPI(title="Fraud-Ops Orchestrator", version="1.0.0", lifespan=lifespan)


@app.get("/health", tags=["Health"])
async def health():
    try:
        await asyncio.to_thread(ping)
        db_ok = True
    except Exception as e:  # noqa: BLE001
        log.error("health DB ping failed: %s", e)
        db_ok = False
    return {"status": "ok" if db_ok else "degraded",
            "service": "fraud-ops-orchestrator", "db": db_ok}


# ---------------------------------------------------------------------------
# Ingestion — TESTING ONLY.
# In production, a separate service writes complaint/instrument/incidents rows
# into Postgres; this orchestrator just polls them. This endpoint exists so you
# can seed rows locally without that other service. (You can also use seed.sql.)
# ---------------------------------------------------------------------------

@app.post("/api/v1/complaints", response_model=IngestAck, tags=["Ingestion"])
async def ingest_complaint(payload: ComplaintIn = Body(...)):
    return await asyncio.to_thread(_ingest, payload)


def _ingest(payload: ComplaintIn) -> IngestAck:
    """TESTING ONLY. Writes ONLY the shared schema.txt tables, exactly as the
    other service would — it does NOT create the sidecar processing row. The
    poller bootstraps that on first pickup, so this faithfully simulates prod."""
    with session_scope() as s:
        existing = s.execute(
            select(Complaint).where(
                Complaint.acknowledgement_no == payload.acknowledgement_no)
        ).scalar_one_or_none()
        if existing:
            log.info("duplicate ingest ack_no=%s", payload.acknowledgement_no)
            return IngestAck(acknowledgement_no=existing.acknowledgement_no,
                             job_id="", status="ALREADY_RECEIVED")

        complaint = Complaint(
            acknowledgement_no=payload.acknowledgement_no,
            sub_category=payload.sub_category,
            status=STATUS_OPEN,
        )
        inst = payload.instrument
        instrument = Instrument(
            requestor=inst.requestor, payer_bank=inst.payer_bank,
            payer_bank_code=inst.payer_bank_code, mode_of_payment=inst.mode_of_payment,
            transaction_type=inst.transaction_type, wallet=inst.wallet,
            payer_mobile_number=inst.payer_mobile_number,
            payer_account_number=inst.payer_account_number,
            state=inst.state, district=inst.district,
        )
        instrument.incidents = [
            Incident(
                merchant_id=i.merchant_id, amount=i.amount,
                disputed_amount=i.disputed_amount, layer=i.layer,
                transaction_date=i.transaction_date, transaction_time=i.transaction_time,
            )
            for i in inst.incidents
        ]
        complaint.instrument = instrument
        s.add(complaint)
        s.commit()
        log.info("ingested ack_no=%s incidents=%d (sidecar created by poller)",
                 payload.acknowledgement_no, len(inst.incidents))
        return IngestAck(acknowledgement_no=payload.acknowledgement_no,
                         job_id="", status="RECEIVED")

@app.post("/process/{ack_no}", tags=["Processing"])
async def process_one(ack_no: str):
    def _run():
        with session_scope() as s:
            c = s.execute(
                select(Complaint).where(Complaint.acknowledgement_no == ack_no)
            ).scalar_one_or_none()
            if c is None:
                return None
            p = s.execute(
                select(ComplaintProcessing).where(
                    ComplaintProcessing.complaint_id == c.id)
            ).scalar_one_or_none()
            if p is None:
                p = ComplaintProcessing(
                    complaint_id=c.id, acknowledgement_no=c.acknowledgement_no,
                    job_id=f"BANKS-{uuid.uuid4()}",
                )
                s.add(p); s.commit()
            pid = p.id
        return processor._process_one(pid)

    state = await asyncio.to_thread(_run)
    if state is None:
        raise HTTPException(404, detail=f"complaint {ack_no} not found")
    return {"acknowledgement_no": ack_no, "state": state}

@app.get("/api/v1/complaints/{ack_no}", tags=["Ingestion"])
async def get_complaint(ack_no: str):
    def _get():
        with session_scope() as s:
            c = s.execute(
                select(Complaint).where(Complaint.acknowledgement_no == ack_no)
            ).scalar_one_or_none()
            if c is None:
                return None
            p = s.execute(
                select(ComplaintProcessing).where(
                    ComplaintProcessing.complaint_id == c.id)
            ).scalar_one_or_none()
            resp = None
            if p and p.job_id:
                resp = s.execute(
                    select(ApiResponseLog).where(ApiResponseLog.job_id == p.job_id)
                ).scalar_one_or_none()
            return {
                "acknowledgement_no": c.acknowledgement_no,
                "complaint_status": c.status,           # shared table (read-only)
                "job_id": p.job_id if p else None,
                "processing_state": p.processing_state if p else "NOT_PICKED_UP",
                "attempt_count": p.attempt_count if p else 0,
                "last_error": p.last_error if p else None,
                "response": resp.response_payload if resp else None,
            }

    data = await asyncio.to_thread(_get)
    if data is None:
        raise HTTPException(404, detail=f"complaint {ack_no} not found")
    return data


@app.post("/debug/cbdoc-search", tags=["Debug"])
async def debug_cbdoc_search(filters: dict = Body(
    ..., examples=[{"OGTRNO": "615213679588"}, {"OGBEAC": "INHSBC500021738001"}])):
    """Run a raw CBDOC/BigQuery search with whatever filters you pass and return
    the row count + the first row's full fields. Use this to discover which
    column holds the account and what value/format CBDOC stores, then set
    BQ_FIELD_ACCOUNT accordingly. Only whitelisted fields are honoured by the
    gateway; unknown fields are ignored (which returns unfiltered rows)."""
    def _run():
        resp = processor.gateway().bigquery_search(filters, limit=20)
        rows = resp.get("rows", []) if isinstance(resp, dict) else []
        return {"filters_sent": filters, "row_count": len(rows),
                "first_row": rows[0] if rows else None}
    return await asyncio.to_thread(_run)
