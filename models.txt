"""
models.py — SQLAlchemy ORM mapped to your schema.txt.

Tables already exist in the DB (created from schema.sql), so we never call
create_all(). GENERATED columns are marked Computed() so they're excluded
from INSERTs. PG ENUM columns are mapped as String — the DB still enforces
the enum constraint; we just don't redeclare the type here.

ADDITIONS to your schema (see migrations.sql) — needed for the polling state
machine and idempotent holds:
  complaints: processing_state, job_id, attempt_count, lease_expires_at, last_error
  new table:  processing_steps  (per-call audit + idempotency keys)
"""
from __future__ import annotations

from datetime import date, datetime, time
from decimal import Decimal
from typing import Optional

from sqlalchemy import (
    BigInteger, Computed, Date, DateTime, ForeignKey, Integer,
    Numeric, String, Text, Time, func,
)
from sqlalchemy.dialects.postgresql import ENUM, JSONB
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, relationship


class Base(DeclarativeBase):
    pass


# Native PG enum types (already created in the DB; create_type=False so
# SQLAlchemy maps to them without trying to CREATE TYPE or cast to VARCHAR).
PaymentMode = ENUM(
    "Credit", "Debit", "UPI", "NEFT", "RTGS", "IMPS",
    name="payment_mode", create_type=False,
)
TransactionType = ENUM(
    "P2P", "P2M", "Collect", "Refund", "Unknown",
    name="transaction_type", create_type=False,
)
FraudSubCategory = ENUM(
    "UPI Related Frauds", "Card Fraud", "Net Banking Fraud", "Wallet Fraud", "Other",
    name="fraud_sub_category", create_type=False,
)
ApiResponseStatus = ENUM(
    "SUCCESS", "FAILURE", "PENDING", "TIMEOUT",
    name="api_response_status", create_type=False,
)


# Coarse external status
STATUS_OPEN        = "OPEN"
STATUS_IN_PROGRESS = "IN_PROGRESS"
STATUS_RESOLVED    = "RESOLVED"
STATUS_REVIEW      = "NEEDS_REVIEW"

# Fine-grained processing state machine
ST_RECEIVED        = "RECEIVED"
ST_CLASSIFIED      = "CLASSIFIED"
ST_ENRICHED        = "ENRICHED"
ST_BALANCE_CHECKED = "BALANCE_CHECKED"
ST_HOLD_APPLIED    = "HOLD_APPLIED"
ST_RECONCILING     = "RECONCILING"
ST_RESOLVED        = "RESOLVED"
ST_NEEDS_REVIEW    = "NEEDS_REVIEW"

TERMINAL_STATES = {ST_RESOLVED, ST_NEEDS_REVIEW}


class Complaint(Base):
    """EXACTLY schema.txt — shared with the other service. DO NOT add columns.
    Treated read-only by this service (we never write to it)."""
    __tablename__ = "complaints"

    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    acknowledgement_no: Mapped[str] = mapped_column(String(20), unique=True)
    sub_category: Mapped[str] = mapped_column(FraudSubCategory)
    status: Mapped[str] = mapped_column(String(20), default=STATUS_OPEN)
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())

    instrument: Mapped["Instrument"] = relationship(
        back_populates="complaint", uselist=False, cascade="all, delete-orphan"
    )
    # 1:1 to our own state table (NOT part of schema.txt).
    processing: Mapped[Optional["ComplaintProcessing"]] = relationship(
        back_populates="complaint", uselist=False, cascade="all, delete-orphan"
    )


class ComplaintProcessing(Base):
    """Sidecar state owned ENTIRELY by this orchestrator (orchestrator_tables.sql).
    Keeps all processing state OUT of the shared `complaints` table so the other
    service's schema is untouched. One row per complaint."""
    __tablename__ = "complaint_processing"

    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    complaint_id: Mapped[int] = mapped_column(
        ForeignKey("complaints.id", ondelete="CASCADE"), unique=True)
    acknowledgement_no: Mapped[str] = mapped_column(String(20))   # denormalized

    processing_state: Mapped[str] = mapped_column(String(30), default=ST_RECEIVED)
    job_id: Mapped[Optional[str]] = mapped_column(String(60))
    attempt_count: Mapped[int] = mapped_column(Integer, default=0)
    lease_expires_at: Mapped[Optional[datetime]] = mapped_column(DateTime(timezone=True))
    last_error: Mapped[Optional[str]] = mapped_column(Text)
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())

    complaint: Mapped["Complaint"] = relationship(back_populates="processing")


class Instrument(Base):
    __tablename__ = "instruments"

    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    complaint_id: Mapped[int] = mapped_column(ForeignKey("complaints.id", ondelete="CASCADE"))

    requestor: Mapped[str] = mapped_column(String(50))
    payer_bank: Mapped[str] = mapped_column(String(100))
    payer_bank_code: Mapped[str] = mapped_column(String(10))
    mode_of_payment: Mapped[str] = mapped_column(PaymentMode)
    transaction_type: Mapped[str] = mapped_column(TransactionType, default="Unknown")
    wallet: Mapped[Optional[str]] = mapped_column(String(50))
    payer_mobile_number: Mapped[str] = mapped_column(String(15))
    payer_account_number: Mapped[str] = mapped_column(String(30))
    state: Mapped[Optional[str]] = mapped_column(String(60))
    district: Mapped[Optional[str]] = mapped_column(String(60))
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())

    complaint: Mapped["Complaint"] = relationship(back_populates="instrument")
    incidents: Mapped[list["Incident"]] = relationship(
        back_populates="instrument", cascade="all, delete-orphan"
    )


class Incident(Base):
    __tablename__ = "incidents"

    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    instrument_id: Mapped[int] = mapped_column(ForeignKey("instruments.id", ondelete="CASCADE"))

    merchant_id: Mapped[str] = mapped_column(String(30))         # the MHA "m" field == RRN
    amount: Mapped[Decimal] = mapped_column(Numeric(12, 2))
    disputed_amount: Mapped[Decimal] = mapped_column(Numeric(12, 2))
    layer: Mapped[int] = mapped_column(Integer, default=0)
    transaction_date: Mapped[date] = mapped_column(Date)
    transaction_time: Mapped[time] = mapped_column(Time)
    transaction_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        Computed("(transaction_date + transaction_time) AT TIME ZONE 'Asia/Kolkata'", persisted=True),
    )
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())

    instrument: Mapped["Instrument"] = relationship(back_populates="incidents")


class ApiResponseLog(Base):
    """Final collated response sent back to MHA — one row per job."""
    __tablename__ = "api_response_logs"

    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    complaint_id: Mapped[Optional[int]] = mapped_column(ForeignKey("complaints.id", ondelete="SET NULL"))
    acknowledgement_no: Mapped[str] = mapped_column(String(20))
    job_id: Mapped[str] = mapped_column(String(60), unique=True)
    response_code: Mapped[str] = mapped_column(String(10))       # "00" = success
    response_message: Mapped[str] = mapped_column(ApiResponseStatus, default="SUCCESS")
    response_payload: Mapped[Optional[dict]] = mapped_column(JSONB)
    is_success: Mapped[bool] = mapped_column(Computed("response_code = '00'", persisted=True))
    requested_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    responded_at: Mapped[Optional[datetime]] = mapped_column(DateTime(timezone=True))
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())


class ProcessingStep(Base):
    """Per-external-call audit + idempotency. NOT in your original schema —
    see migrations.sql. The unique idempotency_key is what makes the
    apply-hold call safe to retry without double-holding funds."""
    __tablename__ = "processing_steps"

    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    complaint_id: Mapped[int] = mapped_column(ForeignKey("complaints.id", ondelete="CASCADE"))
    acknowledgement_no: Mapped[str] = mapped_column(String(20))
    job_id: Mapped[Optional[str]] = mapped_column(String(60))
    step: Mapped[str] = mapped_column(String(40))               # DD_ENQUIRY | HOLD | BQ_TRACE | UPI_BENE
    idempotency_key: Mapped[Optional[str]] = mapped_column(String(200), unique=True)
    status: Mapped[str] = mapped_column(String(20), default="PENDING")  # PENDING|SUCCESS|FAILURE
    request_payload: Mapped[Optional[dict]] = mapped_column(JSONB)
    response_payload: Mapped[Optional[dict]] = mapped_column(JSONB)
    attempt: Mapped[int] = mapped_column(Integer, default=1)
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), server_default=func.now())
