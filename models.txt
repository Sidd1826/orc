"""
models.py — SQLAlchemy 2.0 models matching schema.sql + migration.sql.

We use string columns for status fields (not SQLAlchemy Enums) because the
worker only writes a subset of values and pinning them to Postgres enums
would mean a migration every time we add a state. The Python `Status` class
below is the source of truth for valid values.
"""
from __future__ import annotations

from datetime import date, datetime, time
from decimal import Decimal
from typing import Any, Optional

from sqlalchemy import (
    BigInteger,
    Date,
    DateTime,
    ForeignKey,
    Integer,
    Numeric,
    SmallInteger,
    String,
    Text,
    Time,
)
from sqlalchemy.dialects.postgresql import JSONB
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, relationship


class Base(DeclarativeBase):
    pass


# ---------------------------------------------------------------------------
# Status constants — single source of truth.
# Stored as VARCHAR(20) in the DB, so adding a new one is code-only.
# ---------------------------------------------------------------------------

class ComplaintStatus:
    OPEN                  = "OPEN"
    PROCESSING            = "PROCESSING"
    BALANCE_FETCHED       = "BALANCE_FETCHED"
    HOLD_APPLIED_FULL     = "HOLD_APPLIED_FULL"
    HOLD_APPLIED_PARTIAL  = "HOLD_APPLIED_PARTIAL"
    TRACING               = "TRACING"
    TRACE_COMPLETE        = "TRACE_COMPLETE"
    RESPONSE_READY        = "RESPONSE_READY"
    RESOLVED              = "RESOLVED"
    FAILED_RETRYABLE      = "FAILED_RETRYABLE"
    FAILED_TERMINAL       = "FAILED_TERMINAL"
    MANUAL_REVIEW         = "MANUAL_REVIEW"


# Statuses the sweep is allowed to claim.
CLAIMABLE_STATUSES = (
    ComplaintStatus.OPEN,
    ComplaintStatus.FAILED_RETRYABLE,
)

# Terminal statuses — sweep ignores these.
TERMINAL_STATUSES = (
    ComplaintStatus.RESOLVED,
    ComplaintStatus.FAILED_TERMINAL,
    ComplaintStatus.MANUAL_REVIEW,
)


class ApiResponseStatus:
    SUCCESS = "SUCCESS"
    FAILURE = "FAILURE"
    PENDING = "PENDING"
    TIMEOUT = "TIMEOUT"


# ---------------------------------------------------------------------------
# Tables
# ---------------------------------------------------------------------------

class Complaint(Base):
    __tablename__ = "complaints"

    id:                 Mapped[int]      = mapped_column(BigInteger, primary_key=True)
    acknowledgement_no: Mapped[str]      = mapped_column(String(20), unique=True, nullable=False)
    sub_category:       Mapped[str]      = mapped_column(String, nullable=False)
    status:             Mapped[str]      = mapped_column(String(20), nullable=False, default=ComplaintStatus.OPEN)
    created_at:         Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)
    updated_at:         Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)

    # Worker bookkeeping (added by migration.sql)
    attempt_count:      Mapped[int]                = mapped_column(Integer, nullable=False, default=0)
    last_error:         Mapped[Optional[str]]      = mapped_column(Text, nullable=True)
    locked_at:          Mapped[Optional[datetime]] = mapped_column(DateTime(timezone=True), nullable=True)

    instrument: Mapped["Instrument"] = relationship(back_populates="complaint", uselist=False)


class Instrument(Base):
    __tablename__ = "instruments"

    id:                   Mapped[int]      = mapped_column(BigInteger, primary_key=True)
    complaint_id:         Mapped[int]      = mapped_column(ForeignKey("complaints.id", ondelete="CASCADE"), nullable=False, unique=True)

    requestor:            Mapped[str]      = mapped_column(String(50), nullable=False)
    payer_bank:           Mapped[str]      = mapped_column(String(100), nullable=False)
    payer_bank_code:      Mapped[str]      = mapped_column(String(10), nullable=False)

    mode_of_payment:      Mapped[str]      = mapped_column(String, nullable=False)
    transaction_type:     Mapped[str]      = mapped_column(String, nullable=False, default="Unknown")
    wallet:               Mapped[Optional[str]] = mapped_column(String(50), nullable=True)

    payer_mobile_number:  Mapped[str]      = mapped_column(String(15), nullable=False)
    payer_account_number: Mapped[str]      = mapped_column(String(30), nullable=False)

    state:                Mapped[Optional[str]] = mapped_column(String(60), nullable=True)
    district:             Mapped[Optional[str]] = mapped_column(String(60), nullable=True)

    created_at:           Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)
    updated_at:           Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)

    complaint: Mapped[Complaint]            = relationship(back_populates="instrument")
    incidents: Mapped[list["Incident"]]     = relationship(back_populates="instrument")


class Incident(Base):
    __tablename__ = "incidents"

    id:                 Mapped[int]      = mapped_column(BigInteger, primary_key=True)
    instrument_id:      Mapped[int]      = mapped_column(ForeignKey("instruments.id", ondelete="CASCADE"), nullable=False)

    merchant_id:        Mapped[str]      = mapped_column(String(30), nullable=False)
    amount:             Mapped[Decimal]  = mapped_column(Numeric(12, 2), nullable=False)
    disputed_amount:    Mapped[Decimal]  = mapped_column(Numeric(12, 2), nullable=False)
    layer:              Mapped[int]      = mapped_column(SmallInteger, nullable=False, default=0)

    transaction_date:   Mapped[date]     = mapped_column(Date, nullable=False)
    transaction_time:   Mapped[time]     = mapped_column(Time, nullable=False)
    # transaction_at is a Postgres GENERATED column:
    #   (transaction_date + transaction_time) AT TIME ZONE 'Asia/Kolkata'
    # We mark it read-only with server_default + don't write to it.
    transaction_at:     Mapped[Optional[datetime]] = mapped_column(
        DateTime(timezone=True), nullable=True,
        server_default=None,
    )

    created_at:         Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)
    updated_at:         Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)

    instrument: Mapped[Instrument] = relationship(back_populates="incidents")


class ApiResponseLog(Base):
    __tablename__ = "api_response_logs"

    id:                  Mapped[int]                = mapped_column(BigInteger, primary_key=True)
    complaint_id:        Mapped[Optional[int]]      = mapped_column(ForeignKey("complaints.id", ondelete="SET NULL"), nullable=True)
    acknowledgement_no:  Mapped[str]                = mapped_column(String(20), nullable=False)
    job_id:              Mapped[str]                = mapped_column(String(60), nullable=False)

    response_code:       Mapped[str]                = mapped_column(String(10), nullable=False)
    response_message:    Mapped[str]                = mapped_column(String, nullable=False, default=ApiResponseStatus.SUCCESS)
    response_payload:    Mapped[Optional[dict[str, Any]]] = mapped_column(JSONB, nullable=True)
    # NOTE: is_success is a Postgres GENERATED column (response_code = '00'),
    # so we don't map it for writes. The orchestrator sets response_code='00'
    # on success and a non-'00' string on failure.

    requested_at:        Mapped[datetime]           = mapped_column(DateTime(timezone=True), nullable=False)
    responded_at:        Mapped[Optional[datetime]] = mapped_column(DateTime(timezone=True), nullable=True)
    # response_latency_ms is also a GENERATED column — don't map it.

    created_at:          Mapped[datetime]           = mapped_column(DateTime(timezone=True), nullable=False)
    updated_at:          Mapped[datetime]           = mapped_column(DateTime(timezone=True), nullable=False)


class ComplaintResponse(Base):
    """The final per-complaint JSON the sender service picks up."""
    __tablename__ = "complaint_responses"

    id:                 Mapped[int]                = mapped_column(BigInteger, primary_key=True)
    complaint_id:       Mapped[int]                = mapped_column(ForeignKey("complaints.id", ondelete="CASCADE"), nullable=False, unique=True)
    acknowledgement_no: Mapped[str]                = mapped_column(String(20), nullable=False)
    response_json:      Mapped[dict[str, Any]]     = mapped_column(JSONB, nullable=False)
    status:             Mapped[str]                = mapped_column(String, nullable=False, default="READY")
    created_at:         Mapped[datetime]           = mapped_column(DateTime(timezone=True), nullable=False)
    sent_at:            Mapped[Optional[datetime]] = mapped_column(DateTime(timezone=True), nullable=True)
    sender_attempts:    Mapped[int]                = mapped_column(Integer, nullable=False, default=0)
