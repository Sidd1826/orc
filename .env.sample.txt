# -------- Logging --------
LOG_LEVEL=INFO


# -------- Database (Cloud SQL Postgres via pg8000 + IAM auth) --------
# Local dev: set DB_HOST + DB_PASSWORD, leave INSTANCE_CONNECTION_NAME blank.
# Cloud Run: leave DB_HOST + DB_PASSWORD blank, set INSTANCE_CONNECTION_NAME.
#            The Cloud SQL Python Connector handles auth via ADC + IAM
#            (no DB password — the service account is the DB user).
DB_USER=fraud
DB_NAME=fraud

# --- local-only ---
DB_HOST=localhost
DB_PORT=5432
DB_PASSWORD=

# --- Cloud Run only ---
# Format: <gcp_project>:<region>:<instance>
# e.g.    my-project:asia-south1:fraud-db
INSTANCE_CONNECTION_NAME=

# Use private IP for the Cloud SQL Connector (recommended on Cloud Run with VPC).
# Set to "true" when the Cloud SQL instance has private IP enabled and the
# Cloud Run service is attached to the same VPC connector.
PRIVATE_IP=false

# Pool sizing — keep low; volume is ~100/day
DB_POOL_SIZE=5
DB_MAX_OVERFLOW=2


# -------- HSBC MHA Gateway (the sibling Cloud Run service) --------
GATEWAY_BASE_URL=https://hsbc-mha-gateway-xxxxx.a.run.app

# If true, fetch an ID token and pass it as Authorization: Bearer for
# Cloud Run service-to-service auth. Set false for unauthenticated dev/UAT.
GATEWAY_USE_IAM_AUTH=true

# Network
HTTP_CONNECT_TIMEOUT=10
HTTP_READ_TIMEOUT=60


# -------- Worker behaviour --------
# Max complaints claimed per sweep tick.
SWEEP_BATCH_SIZE=20

# A complaint is auto-retried this many times before moving to FAILED_TERMINAL.
MAX_ATTEMPTS=5

# Hold expiry — kept long so a hold persists until ops actions it.
HOLD_EXPIRY_DATE=2099-12-31

# "Fresh" vs "historic" cutoff. < 24h => transaction-history; else BigQuery.
FRESH_WINDOW_HOURS=24

# Hard cap on how many BQ-traced debit rows we'll pull when reconciling.
TRACE_MAX_ROWS=500


# -------- Server --------
PORT=8000
