"""
config.py — central config for the fraud-ops orchestrator.

Loads from .env locally (python-dotenv); on Cloud Run values come from
--set-env-vars / --set-secrets. No secrets are hard-coded here.
"""
from __future__ import annotations

import os

try:
    from dotenv import load_dotenv
    load_dotenv(override=False)
except ImportError:
    pass


def _env(name: str, default: str = "") -> str:
    return os.getenv(name, default).strip()


def _env_int(name: str, default: int) -> int:
    raw = os.getenv(name)
    return int(raw) if raw else default


def _env_float(name: str, default: float) -> float:
    raw = os.getenv(name)
    return float(raw) if raw else default


def _env_bool(name: str, default: bool = False) -> bool:
    raw = os.getenv(name)
    return raw.strip().lower() == "true" if raw else default


# --- Postgres (host/user/password — same as your DBeaver connection) ---
# Local/TCP : PG_HOST=localhost (or the DB's IP)
# Cloud Run : PG_HOST=/cloudsql/PROJECT:REGION:INSTANCE  (unix socket)
PG_HOST     = _env("PG_HOST", "localhost")
PG_PORT     = _env("PG_PORT", "5432")
PG_USER     = _env("PG_USER", "")
PG_PASSWORD = _env("PG_PASSWORD", "")
PG_DB       = _env("PG_DB", "")

# --- Upstream gateway ---
GATEWAY_BASE_URL        = _env("GATEWAY_BASE_URL", "http://localhost:8000")
GATEWAY_TIMEOUT_CONNECT = _env_float("GATEWAY_TIMEOUT_CONNECT", 10.0)
GATEWAY_TIMEOUT_READ    = _env_float("GATEWAY_TIMEOUT_READ", 60.0)
USE_OIDC_AUTH           = _env_bool("USE_OIDC_AUTH", False)
GATEWAY_AUDIENCE        = _env("GATEWAY_AUDIENCE", GATEWAY_BASE_URL)

# --- Processing ---
FRAUD_AGE_THRESHOLD_HOURS = _env_int("FRAUD_AGE_THRESHOLD_HOURS", 24)
BATCH_SIZE                = _env_int("BATCH_SIZE", 10)
LEASE_SECONDS             = _env_int("LEASE_SECONDS", 300)
MAX_ATTEMPTS              = _env_int("MAX_ATTEMPTS", 5)
MAX_TRACE_LAYERS          = _env_int("MAX_TRACE_LAYERS", 5)

# --- Upstream JSON field paths (CONFIRM against real HSBC contracts) ---
# --- Balance enquiry: path to the balance amount in the gateway response.
#     The gateway wraps the HSBC body under "data", so the real shape is
#     data.accounts[].currentBalance.amount. Dotted path supports list indices.
#     If the path misses, the code deep-searches for BALANCE_FIELD.amount and
#     logs the path it found, so you can set BALANCE_PATH precisely.
BALANCE_PATH  = _env("BALANCE_PATH", "data.accounts.0.availableBalance.amount")
BALANCE_FIELD = _env("BALANCE_FIELD", "availableBalance")   # deep-search fallback target

# Whether to auto-detect UPI transactions in CBDOC rows (IB Debit >24h path)
# and call the UPI API for bene details. Set false to disable detection entirely.
UPI_DETECT = _env("UPI_DETECT", "true").lower() == "true"

# UPI secure-call systemId (from the working UPI contract). Numeric.
UPI_SYSTEM_ID = _env_int("UPI_SYSTEM_ID", 403)

# --- Transaction enquiry: path to the transaction list (also under "data") ---
TXN_DETAILS_PATH = _env("TXN_DETAILS_PATH", "accountInformation.transactionDetails")

# --- Apply-hold mandatory fields ---
HOLD_TYPE      = _env("HOLD_TYPE", "F")            # per your sample (F)
HOLD_TILL_DATE = _env("HOLD_TILL_DATE", "2099-12-31")
HOLD_CURRENCY  = _env("HOLD_CURRENCY", "INR")

# --- BigQuery (CBDOC) field mapping (confirmed against a real CBDOC row) ---
BQ_FIELD_REF          = _env("BQ_FIELD_REF", "LPUTRN")     # UTR/RRN (NOT OGTRNO)
BQ_FIELD_AMOUNT       = _env("BQ_FIELD_AMOUNT", "OGPYAM")
BQ_FIELD_DATE         = _env("BQ_FIELD_DATE", "OGCPDT")
BQ_FIELD_TIME         = _env("BQ_FIELD_TIME", "OGCPTM")
# Real transaction timestamp column, format 'YYYY-MM-DD-HH.MM.SS.ffffff' (IST).
# Used to compare against the MHA fraud time. Falls back to OGCPDT/OGCPTM if empty.
BQ_FIELD_TIMESTAMP    = _env("BQ_FIELD_TIMESTAMP", "OGTMZ1")
# Account column. Values are stored with a leading '/', e.g. '/INHSBC006361141001'.
# BQ_ACCOUNT_PREFIX is prepended to the account when filtering (the gateway does
# exact-equality), and stripped when reading values client-side.
BQ_FIELD_ACCOUNT      = _env("BQ_FIELD_ACCOUNT", "OGBEAC")
BQ_ACCOUNT_PREFIX     = _env("BQ_ACCOUNT_PREFIX", "/")
BQ_FIELD_BENE_ACCOUNT = _env("BQ_FIELD_BENE_ACCOUNT", "OGBEAC")
CBDOC_MAX_ROWS        = _env_int("CBDOC_MAX_ROWS", 500)

# Optional debit/credit indicator in CBDOC. If your table mixes debits and
# credits, set BQ_DEBIT_INDICATOR_FIELD to the column and BQ_DEBIT_INDICATOR_VALUE
# to the value that marks a debit (e.g. field="OGDRCR", value="D"). Leave the
# field empty if the table is debit-only (OG* = outgoing payments).
BQ_DEBIT_INDICATOR_FIELD = _env("BQ_DEBIT_INDICATOR_FIELD", "")
BQ_DEBIT_INDICATOR_VALUE = _env("BQ_DEBIT_INDICATOR_VALUE", "D")

# When true, the CBDOC account search also filters by the fraud date (BQ_FIELD_DATE)
# so it returns that day's transactions; the caller then keeps those after the
# fraud TIME. Set false to pull all account rows and filter purely client-side.
CBDOC_FILTER_BY_DATE = _env("CBDOC_FILTER_BY_DATE", "false").lower() == "true"
