"""
logger.py — Centralized logging setup.

All modules call `get_logger(__name__)`. The first call configures the root
logger; subsequent calls just return a named child. This keeps log format
consistent across the whole worker.

Cloud Run captures stdout/stderr automatically — no file handlers needed.
"""
from __future__ import annotations

import logging
import os
import sys

_LOG_LEVEL  = os.getenv("LOG_LEVEL", "INFO").upper()
_LOG_FORMAT = "%(asctime)s [%(levelname)s] %(name)s - %(message)s"

_configured = False


def _configure_root_once() -> None:
    """Configure root logger exactly once per process."""
    global _configured
    if _configured:
        return

    handler = logging.StreamHandler(sys.stdout)
    handler.setFormatter(logging.Formatter(_LOG_FORMAT))

    root = logging.getLogger()
    root.handlers = [handler]
    root.setLevel(_LOG_LEVEL)

    for noisy in ("httpx", "httpcore", "urllib3", "asyncio", "sqlalchemy.engine"):
        logging.getLogger(noisy).setLevel(os.getenv(f"LOG_LEVEL_{noisy.upper()}", "WARNING"))

    _configured = True


def get_logger(name: str) -> logging.Logger:
    _configure_root_once()
    return logging.getLogger(name)
