"""
response_builder.py — Assemble the final JSON for the sender service.

The exact shape MHA expects is TBD (sample/spec pending). This builder is
intentionally structured so every section of the work the orchestrator did is
preserved and easy to reshape later. When the spec arrives, only this file
changes — the orchestrator keeps shovelling the same intermediate dicts in.
"""
from __future__ import annotations

from datetime import datetime, timezone
from decimal import Decimal
from typing import Any, Optional

from models import Complaint, Incident, Instrument


def _money(d: Optional[Decimal]) -> Optional[str]:
    return f"{d:.2f}" if d is not None else None


def build_final_response(
    *,
    complaint:           Complaint,
    instrument:          Instrument,
    incidents:           list[Incident],
    total_disputed:      Decimal,
    available_balance:   Optional[Decimal],
    hold_outcome:        dict[str, Any],     # {"status": "FULL"|"PARTIAL"|"NONE", "amount": "..", "reference": ".."}
    transaction_details: list[dict[str, Any]],
    trace_rows:          list[dict[str, Any]],
    reconciliation:      dict[str, Any],     # {"reconciled": bool, "amount_reconciled": "..", "shortfall": ".."}
) -> dict[str, Any]:
    """Build the final JSON document for one resolved complaint."""
    now = datetime.now(timezone.utc).isoformat()

    return {
        "acknowledgementNo": complaint.acknowledgement_no,
        "complaintId":       complaint.id,
        "subCategory":       complaint.sub_category,
        "generatedAt":       now,

        "instrument": {
            "requestor":          instrument.requestor,
            "payerBank":          instrument.payer_bank,
            "payerBankCode":      instrument.payer_bank_code,
            "modeOfPayment":      instrument.mode_of_payment,
            "transactionType":    instrument.transaction_type,
            "wallet":             instrument.wallet,
            "payerMobileNumber":  instrument.payer_mobile_number,
            "payerAccountNumber": instrument.payer_account_number,
            "state":              instrument.state,
            "district":           instrument.district,
        },

        "incidents": [
            {
                "incidentId":      i.id,
                "merchantId":      i.merchant_id,
                "amount":          _money(i.amount),
                "disputedAmount":  _money(i.disputed_amount),
                "layer":           i.layer,
                "transactionAt":   i.transaction_at.isoformat() if i.transaction_at else None,
            }
            for i in incidents
        ],

        "summary": {
            "totalDisputedAmount": _money(total_disputed),
            "availableBalance":    _money(available_balance),
        },

        "hold": hold_outcome,

        "transactionDetails": transaction_details,

        "trace": {
            "rows":         trace_rows,
            "rowCount":     len(trace_rows),
        },

        "reconciliation": reconciliation,
    }
