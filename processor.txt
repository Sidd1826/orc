r"""
processor.py — the orchestration engine.

Trigger model: NO queue. Cloud Scheduler hits /process-batch every 1-2 min;
this module claims a batch of un-finished complaints with
`FOR UPDATE SKIP LOCKED`, leases them, and runs a resumable state machine.

State machine (persisted on complaint_processing.processing_state — a SIDECAR
table this service owns; the shared `complaints` table is never written):

  RECEIVED -> CLASSIFIED -> ENRICHED -> BALANCE_CHECKED
           -> HOLD_APPLIED -> RECONCILING -> RESOLVED
                                          \-> NEEDS_REVIEW (terminal, human)

Money rule (from your prose):
  D = total disputed amount, B = available balance.
  B >= D  -> hold D, done.
  B <  D  -> hold B, remaining R = D - B, then trace R through CBDOC/BigQuery,
             collecting every debited transaction until reconciled (or layers
             exhausted -> NEEDS_REVIEW with whatever was collected).

Idempotency: apply-hold is guarded by a UNIQUE idempotency_key in
processing_steps, so a retry never double-holds a customer's funds.
"""

_gateway: Optional[GatewayClient] = None


def gateway() -> GatewayClient:
    global _gateway
    if _gateway is None:
        _gateway = GatewayClient()
    return _gateway


def _now() -> datetime:
    return datetime.now(timezone.utc)


def _dig(d: Any, path: str) -> Any:
    """Read a dotted path out of a nested dict/list; None if missing.
    Numeric path segments index into lists, e.g. 'accounts.0.currentBalance.amount'."""
    cur = d
    for part in path.split("."):
        if isinstance(cur, list):
            try:
                cur = cur[int(part)]
            except (ValueError, IndexError):
                return None
        elif isinstance(cur, dict) and part in cur:
            cur = cur[part]
        else:
            return None
    return cur


def _usable_number(x: Any) -> bool:
    """True if x is a number or a non-empty numeric string."""
    if isinstance(x, bool):
        return False
    if isinstance(x, (int, float)):
        return True
    if isinstance(x, str):
        try:
            Decimal(x.strip())
            return x.strip() != ""
        except Exception:
            return False
    return False


def _deep_find_balance(obj: Any, container_key: str, amount_key: str = "amount",
                       _path: str = "") -> tuple[Any, str]:
    """Find the balance anywhere in a nested dict/list. Accepts `container_key`
    as EITHER a scalar (e.g. data.currentBalance = "76.21") OR an object
    (e.g. ...currentBalance.amount = 76.21). Returns (value, dotted_path)."""
    if isinstance(obj, dict):
        if container_key in obj:
            v = obj[container_key]
            if isinstance(v, dict) and amount_key in v and _usable_number(v[amount_key]):
                return v[amount_key], f"{_path}.{container_key}.{amount_key}".lstrip(".")
            if _usable_number(v):
                return v, f"{_path}.{container_key}".lstrip(".")
        for k, val in obj.items():
            r, fp = _deep_find_balance(val, container_key, amount_key,
                                       f"{_path}.{k}".lstrip("."))
            if r is not None:
                return r, fp
    elif isinstance(obj, list):
        for i, val in enumerate(obj):
            r, fp = _deep_find_balance(val, container_key, amount_key,
                                       f"{_path}.{i}".lstrip("."))
            if r is not None:
                return r, fp
    return None, ""


def _coerce_balance(x: Any) -> Optional[Decimal]:
    if x is None:
        return None
    try:
        return Decimal(str(x).strip())
    except Exception:
        return None


def _shape(obj: Any, depth: int = 3) -> Any:
    """Keys-only structural description (NO values) — safe to log/show."""
    if isinstance(obj, dict):
        if depth <= 0:
            return {k: "..." for k in obj}
        return {k: _shape(v, depth - 1) for k, v in obj.items()}
    if isinstance(obj, list):
        if not obj:
            return []
        head = _shape(obj[0], depth - 1)
        return [head, f"...(+{len(obj) - 1} more)"] if len(obj) > 1 else [head]
    return type(obj).__name__

# Batch claim (the polling tick)

def _bootstrap_processing_rows(s: Session) -> int:
    """Create a complaint_processing row for any complaint that doesn't have one
    yet (i.e. rows the OTHER service just inserted). Does NOT touch complaints.
    INSERT ... SELECT ... ON CONFLICT DO NOTHING is race-safe across instances.
    """
    res = s.execute(text(
        """
        INSERT INTO complaint_processing (complaint_id, acknowledgement_no, processing_state)
        SELECT c.id, c.acknowledgement_no, 'RECEIVED'
        FROM complaints c
        LEFT JOIN complaint_processing p ON p.complaint_id = c.id
        WHERE p.id IS NULL
        ON CONFLICT (complaint_id) DO NOTHING
        """
    ))
    s.commit()
    return res.rowcount or 0


def claim_batch(limit: int) -> list[int]:
    """Bootstrap missing sidecar rows, then claim up to `limit` non-terminal,
    lease-free processing rows. Returns the claimed complaint_processing ids.
    Nothing in the shared `complaints` table is written."""
    now = _now()
    with session_scope() as s:
        _bootstrap_processing_rows(s)

        rows = s.execute(
            select(ComplaintProcessing)
            .where(
                ComplaintProcessing.processing_state.notin_(tuple(TERMINAL_STATES)),
                ComplaintProcessing.attempt_count < config.MAX_ATTEMPTS,
                or_(ComplaintProcessing.lease_expires_at.is_(None),
                    ComplaintProcessing.lease_expires_at < now),
            )
            .order_by(ComplaintProcessing.created_at)
            .limit(limit)
            .with_for_update(skip_locked=True)
        ).scalars().all()

        lease = now + timedelta(seconds=config.LEASE_SECONDS)
        ids = []
        for p in rows:
            p.lease_expires_at = lease
            if not p.job_id:
                p.job_id = f"BANKS-{uuid.uuid4()}"
            ids.append(p.id)
        s.commit()
        if ids:
            log.info("claimed %d complaints", len(ids))
        return ids


def process_batch() -> dict[str, Any]:
    """One polling tick: claim a batch and process each in its own transaction."""
    ids = claim_batch(config.BATCH_SIZE)
    processed = []
    for pid in ids:
        state = _process_one(pid)
        processed.append({"processing_id": pid, "state": state})
    return {"claimed": len(ids), "processed": processed}


def _process_one(processing_id: int) -> str:
    """Run the state machine for one complaint_processing row. Failures are
    recorded and the row is left for the next tick (until MAX_ATTEMPTS)."""
    try:
        with session_scope() as s:
            p = s.get(ComplaintProcessing, processing_id)
            if p is None or p.processing_state in TERMINAL_STATES:
                return p.processing_state if p else "MISSING"
            _run_state_machine(s, p)
            return p.processing_state
    except Exception as e:                      # noqa: BLE001 - record & defer
        log.exception("processing failed for processing_id=%s: %s", processing_id, e)
        _record_failure(processing_id, str(e))
        with session_scope() as s:
            p = s.get(ComplaintProcessing, processing_id)
            return p.processing_state if p else "MISSING"


def _record_failure(processing_id: int, err: str) -> None:
    with session_scope() as s:
        p = s.get(ComplaintProcessing, processing_id)
        if p is None:
            return
        p.attempt_count = (p.attempt_count or 0) + 1
        p.last_error = err[:2000]
        if p.attempt_count >= config.MAX_ATTEMPTS:
            p.processing_state = ST_NEEDS_REVIEW
            p.lease_expires_at = None
            log.error("processing_id=%s exhausted retries -> NEEDS_REVIEW", processing_id)
        # else: lease left in place; next tick re-picks after it expires (backoff)
        s.commit()


# ===========================================================================
# State machine
# ===========================================================================

def _run_state_machine(s: Session, p: ComplaintProcessing) -> None:
    c = p.complaint
    instrument = c.instrument if c else None
    incidents = instrument.incidents if instrument else []
    if not instrument or not incidents:
        _to_review(s, p, "no instrument/incidents on complaint")
        return

    cls = classify(instrument, incidents, c.sub_category)
    account = instrument.payer_account_number
    total_disputed = sum((i.disputed_amount for i in incidents), Decimal("0"))

    # Advance one step at a time; commit after each so retries resume here.
    guard = 0
    while p.processing_state not in TERMINAL_STATES and guard < 20:
        guard += 1
        st = p.processing_state
        log.info("[%s] state=%s", p.acknowledgement_no, st)

        if st == ST_RECEIVED:
            p.processing_state = ST_CLASSIFIED

        elif st == ST_CLASSIFIED:
            _enrich(s, p, cls, instrument, incidents)
            p.processing_state = ST_ENRICHED

        elif st == ST_ENRICHED:
            # UPI-debit fraud only needs beneficiary details -> no hold.
            if cls.channel == CHANNEL_UPI and cls.direction == DIR_DEBIT:
                _finalize_bene_only(s, p, incidents)
                p.processing_state = ST_RESOLVED
                p.lease_expires_at = None
                s.add(p); s.commit()
                break
            # IB Debit: DD-validated UTR -> CBDOC bene -> respond, NO hold.
            # If the UTR wasn't validated, return 'miscellaneous' to MHA.
            if cls.direction == DIR_DEBIT:
                if not _rrn_validated(s, p, incidents):
                    _finalize_miscellaneous(s, p, incidents,
                                            "UTR not found in DD transaction enquiry")
                    p.processing_state = ST_NEEDS_REVIEW
                    p.last_error = "UTR not validated in DD enquiry (miscellaneous)"
                else:
                    _finalize_cbdoc_bene(s, p, incidents)
                    p.processing_state = ST_RESOLVED
                p.lease_expires_at = None
                s.add(p); s.commit()
                break
            # CREDIT: RRN must have validated in the DD enquiry. If not, return a
            # 'miscellaneous' response to MHA and stop — no balance, no hold.
            if cls.direction == DIR_CREDIT and not _rrn_validated(s, p, incidents):
                _finalize_miscellaneous(s, p, incidents,
                                        "RRN not found in DD transaction enquiry")
                p.processing_state = ST_NEEDS_REVIEW
                p.last_error = "RRN not validated in DD enquiry (miscellaneous)"
                p.lease_expires_at = None
                s.add(p); s.commit()
                break
            p.processing_state = ST_BALANCE_CHECKED

        elif st == ST_BALANCE_CHECKED:
            balance = _available_balance(s, p, account)
            held, remaining = _apply_holds(s, p, instrument, total_disputed, balance)
            _stash(s, p, "REMAINING", {"value": str(remaining), "held": str(held)})
            p.processing_state = ST_HOLD_APPLIED

        elif st == ST_HOLD_APPLIED:
            remaining = Decimal(_unstash(s, p, "REMAINING").get("value", "0"))
            if remaining > 0:
                p.processing_state = ST_RECONCILING
            else:
                _finalize(s, p, instrument, incidents, traced_txns=[])
                p.processing_state = ST_RESOLVED

        elif st == ST_RECONCILING:
            remaining = Decimal(_unstash(s, p, "REMAINING").get("value", "0"))
            traced, reconciled = _trace_and_collect(s, p, incidents, remaining)
            _finalize(s, p, instrument, incidents, traced_txns=traced)
            p.processing_state = ST_RESOLVED if reconciled else ST_NEEDS_REVIEW
            if not reconciled:
                p.last_error = "CBDOC account debits did not reconcile the disputed amount"
            p.lease_expires_at = None
            s.add(p); s.commit()
            break

        s.add(p)
        s.commit()

    if p.processing_state == ST_RESOLVED:
        p.lease_expires_at = None
        s.commit()


def _to_review(s: Session, p: ComplaintProcessing, reason: str) -> None:
    p.processing_state = ST_NEEDS_REVIEW
    p.last_error = reason
    p.lease_expires_at = None
    s.commit()
    log.error("[%s] -> NEEDS_REVIEW: %s", p.acknowledgement_no, reason)


# ===========================================================================
# Steps (each cached in processing_steps so repeats don't re-hit upstream)
# ===========================================================================

def _step_get(s: Session, p: ComplaintProcessing, step: str,
              idem: Optional[str] = None) -> Optional[ProcessingStep]:
    key = idem or f"{step}:{p.acknowledgement_no}"
    return s.execute(
        select(ProcessingStep).where(ProcessingStep.idempotency_key == key)
    ).scalar_one_or_none()


def _step_record(s: Session, p: ComplaintProcessing, step: str, idem: str,
                 request: dict, response: dict, status: str = "SUCCESS") -> ProcessingStep:
    row = ProcessingStep(
        complaint_id=p.complaint_id, acknowledgement_no=p.acknowledgement_no,
        job_id=p.job_id, step=step, idempotency_key=idem, status=status,
        request_payload=request, response_payload=response,
    )
    s.add(row)
    try:
        s.flush()
    except IntegrityError:          # concurrent insert — reuse existing
        s.rollback()
        return _step_get(s, p, step, idem)   # type: ignore[return-value]
    return row


def _stash(s: Session, p: ComplaintProcessing, name: str, data: dict) -> None:
    idem = f"STASH:{name}:{p.acknowledgement_no}"
    existing = _step_get(s, p, name, idem)
    if existing:
        existing.response_payload = data
        s.flush()
    else:
        _step_record(s, p, f"STASH_{name}", idem, {}, data)


def _unstash(s: Session, p: ComplaintProcessing, name: str) -> dict:
    idem = f"STASH:{name}:{p.acknowledgement_no}"
    row = _step_get(s, p, name, idem)
    return (row.response_payload or {}) if row else {}


def _enrich(s: Session, p: ComplaintProcessing, cls, instrument: Instrument,
            incidents: list[Incident]) -> None:
    """Fetch beneficiary / transaction context.

    Scenarios:
      UPI Debit any age   → UPI API only (no hold)
      CREDIT (any age)    → DD transaction enquiry FIRST to validate the RRN.
                            If RRN not found in any narrative → caller stops and
                            returns a 'miscellaneous' response to MHA (no hold).
                            If found: balance → hold; and if >24h + partial, the
                            reconcile step traces CBDOC debits after the fraud time.
      IB Debit <24h       → DD transaction enquiry → balance → hold
      IB Debit >24h       → CBDOC locate → UPI API if txn is UPI → balance → hold
    """
    account = instrument.payer_account_number

    # --- UPI Debit: short-circuit, bene lookup only, no hold ---
    if cls.channel == CHANNEL_UPI and cls.direction == DIR_DEBIT:
        for inc in incidents:
            idem = f"UPI:{p.acknowledgement_no}:{inc.merchant_id}"
            if _step_get(s, p, "UPI_BENE", idem):
                continue
            ts = f"{inc.transaction_date} {inc.transaction_time}"
            resp = gateway().upi_bene_lookup(
                rrn=inc.merchant_id, account_number=account,
                amount=str(inc.amount), transaction_timestamp=ts,
            )
            _step_record(s, p, "UPI_BENE", idem, {"rrn": inc.merchant_id}, resp)
        return

    # --- CREDIT (any age): validate RRN in the live DD transaction enquiry FIRST ---
    if cls.direction == DIR_CREDIT:
        from_date = min(i.transaction_date for i in incidents).isoformat()
        resp = _txn_enquiry(s, p, account, from_date)
        details = _dig(resp, config.TXN_DETAILS_PATH) or []
        if not details:
            # Path yielded nothing — either the real API nests the txn list
            # elsewhere, or the date filter returned no rows. Self-heal: deep-find
            # the list, and log the real shape so the path can be corrected.
            details = _deep_find_txn_list(resp)
            log.warning("[%s] DD enquiry: 0 txns at TXN_DETAILS_PATH='%s'. "
                        "deep-find got %d. response keys=%s",
                        p.acknowledgement_no, config.TXN_DETAILS_PATH,
                        len(details), _shape(resp))
        log.info("[%s] DD enquiry returned %d transactions (fromDate=%s)",
                 p.acknowledgement_no, len(details), from_date)
        for inc in incidents:
            idem = f"DDTXN:{p.acknowledgement_no}:{inc.merchant_id}"
            if _step_get(s, p, "DD_TXN", idem):
                continue
            matched = _find_txn_by_rrn(details, inc.merchant_id)
            _step_record(s, p, "DD_TXN", idem,
                         {"rrn": inc.merchant_id},
                         {"found": matched is not None, "matched": matched or {}})
            if matched is None:
                log.warning("[%s] RRN=%s not found in DD enquiry narratives",
                            p.acknowledgement_no, inc.merchant_id)
                _log_rrn_miss(p, details, inc.merchant_id)
            else:
                log.info("[%s] RRN=%s MATCHED. transaction=%s",
                         p.acknowledgement_no, inc.merchant_id,
                         json.dumps(matched, default=str))
        return

    # --- IB Debit: validate UTR in DD enquiry, then CBDOC (by UTR) for bene.
    #     No hold — like UPI Debit, but bene comes from CBDOC not the UPI API. ---
    if cls.direction == DIR_DEBIT:
        from_date = min(i.transaction_date for i in incidents).isoformat()
        resp = _txn_enquiry(s, p, account, from_date)
        details = _dig(resp, config.TXN_DETAILS_PATH) or []
        if not details:
            details = _deep_find_txn_list(resp)
            log.warning("[%s] DD enquiry: 0 txns at TXN_DETAILS_PATH='%s'. "
                        "deep-find got %d. response keys=%s",
                        p.acknowledgement_no, config.TXN_DETAILS_PATH,
                        len(details), _shape(resp))
        log.info("[%s] DD enquiry returned %d transactions (fromDate=%s)",
                 p.acknowledgement_no, len(details), from_date)
        for inc in incidents:
            ddidem = f"DDTXN:{p.acknowledgement_no}:{inc.merchant_id}"
            if not _step_get(s, p, "DD_TXN", ddidem):
                matched = _find_txn_by_rrn(details, inc.merchant_id)
                _step_record(s, p, "DD_TXN", ddidem, {"rrn": inc.merchant_id},
                             {"found": matched is not None, "matched": matched or {}})
                if matched is None:
                    log.warning("[%s] UTR=%s not found in DD enquiry narratives",
                                p.acknowledgement_no, inc.merchant_id)
                    _log_rrn_miss(p, details, inc.merchant_id)
                    continue
                log.info("[%s] UTR=%s MATCHED in DD enquiry", p.acknowledgement_no, inc.merchant_id)

            # UTR validated -> fetch bene details from CBDOC by the UTR/RRN.
            bqidem = f"CBDOCBENE:{p.acknowledgement_no}:{inc.merchant_id}"
            if _step_get(s, p, "CBDOC_BENE", bqidem):
                continue
            bq = gateway().bigquery_search({config.BQ_FIELD_REF: inc.merchant_id}, limit=5)
            _step_record(s, p, "CBDOC_BENE", bqidem, {config.BQ_FIELD_REF: inc.merchant_id}, bq)
            n = len(_rows(bq))
            log.info("[%s] CBDOC bene lookup by UTR=%s -> %d rows",
                     p.acknowledgement_no, inc.merchant_id, n)
        return

    # --- (unreachable: directions handled above) fallback CBDOC locate ---
    for inc in incidents:
        _bq_locate(s, p, account, inc)


def _is_upi_txn(bq_row: dict, inc: Incident) -> bool:
    """Decide if a CBDOC row represents a UPI transaction.
    Checks narratives (OGNARx) for 'UPI' keyword or a UPI VPA pattern (x@y).
    Also checks the incident's merchant_id — UPI RRNs are typically 12 digits.
    Override by setting UPI_DETECT=false in env to skip this check entirely."""
    if not config.UPI_DETECT:
        return False

    # Check CBDOC narrative fields for UPI indicators
    for field in ("OGNAR1", "OGNAR2", "OGNAR3", "OGNAR4", "OGTMZ1"):
        val = str(bq_row.get(field) or "").upper()
        if "UPI" in val:
            log.info("UPI detected in CBDOC field %s='%s'", field, val)
            return True
        # UPI VPA pattern: something@something (e.g. user@oksbi)
        if "@" in val and len(val.split("@")) == 2:
            log.info("UPI VPA detected in CBDOC field %s='%s'", field, val)
            return True

    # 12-digit numeric RRN is a strong UPI indicator
    rrn = str(inc.merchant_id or "").strip()
    if rrn.isdigit() and len(rrn) == 12:
        log.info("12-digit RRN %s suggests UPI transaction", rrn)
        return True

    return False


def _upi_bene_if_required(s: Session, p: ComplaintProcessing, inc: Incident,
                           account: str) -> None:
    """Call UPI bene lookup for an IB Debit >24h that was identified as UPI.
    Idempotent — skipped if already done for this incident."""
    idem = f"UPI:{p.acknowledgement_no}:{inc.merchant_id}"
    if _step_get(s, p, "UPI_BENE", idem):
        log.info("[%s] UPI bene already fetched for RRN=%s (skip)",
                 p.acknowledgement_no, inc.merchant_id)
        return

    log.info("[%s] IB Debit >24h: calling UPI API for RRN=%s",
             p.acknowledgement_no, inc.merchant_id)
    ts = f"{inc.transaction_date} {inc.transaction_time}"
    try:
        resp = gateway().upi_bene_lookup(
            rrn=inc.merchant_id, account_number=account,
            amount=str(inc.amount), transaction_timestamp=ts,
        )
        _step_record(s, p, "UPI_BENE", idem, {"rrn": inc.merchant_id}, resp)
        log.info("[%s] UPI bene fetched for RRN=%s", p.acknowledgement_no, inc.merchant_id)
    except GatewayError as e:
        # UPI API is "if required" — log but don't fail the whole complaint
        log.warning("[%s] UPI bene lookup failed for RRN=%s (non-fatal): %s",
                    p.acknowledgement_no, inc.merchant_id, e)
        _step_record(s, p, "UPI_BENE", idem,
                     {"rrn": inc.merchant_id},
                     {"error": str(e)}, status="FAILURE")


def _txn_enquiry(s: Session, p: ComplaintProcessing, account: str, from_date: str) -> dict:
    idem = f"TXN:{p.acknowledgement_no}:{account}"
    existing = _step_get(s, p, "TXN_ENQUIRY", idem)
    if existing and existing.status == "SUCCESS":
        return existing.response_payload or {}
    resp = gateway().transaction_history(account, from_date=from_date)
    _step_record(s, p, "TXN_ENQUIRY", idem,
                 {"account": account, "fromDate": from_date}, resp)
    return resp


def _narratives(txn: dict) -> list[str]:
    """Collect every narrative string from a transaction, robust to shape.

    Handles all of:
      transactionDescription: [{transactionNarrative: "..."}, ...]
      transactionDescription: ["...", "..."]            (plain strings)
      transactionNarrative:    "..."  or  ["...", ...]   (directly on txn)
      any nested dict/list containing 'narrative' keys
    """
    out: list[str] = []

    def _add(v):
        if isinstance(v, str) and v.strip():
            out.append(v.strip())

    def _walk(node):
        if isinstance(node, dict):
            for k, v in node.items():
                if "narrative" in k.lower():
                    if isinstance(v, str):
                        _add(v)
                    elif isinstance(v, list):
                        for item in v:
                            _add(item) if isinstance(item, str) else _walk(item)
                    else:
                        _walk(v)
                elif isinstance(v, (dict, list)):
                    _walk(v)
        elif isinstance(node, list):
            for item in node:
                _walk(item)

    # Primary, known shape first (keeps order predictable)
    desc = txn.get("transactionDescription")
    if isinstance(desc, list):
        for d in desc:
            if isinstance(d, dict):
                _add(d.get("transactionNarrative"))
            elif isinstance(d, str):
                _add(d)

    # Then sweep the whole txn for anything missed / differently nested
    _walk(txn)

    # de-dup, preserve order
    seen = set()
    uniq = []
    for n in out:
        if n not in seen:
            seen.add(n)
            uniq.append(n)
    return uniq


def _norm_ref(s: str) -> str:
    """Normalise a reference for comparison: drop all whitespace, uppercase."""
    return re.sub(r"\s+", "", str(s or "")).upper()


def _deep_find_txn_list(resp: Any) -> list[dict]:
    """Find the transaction-details list anywhere in the response, regardless of
    the exact path. A match = a list of dicts where the dicts contain
    'transactionDescription' (or 'transactionNarrative' nested). Lets us recover
    when TXN_DETAILS_PATH doesn't match the real API shape."""
    found: list[dict] = []

    def _looks_like_txn(d: dict) -> bool:
        if not isinstance(d, dict):
            return False
        if "transactionDescription" in d:
            return True
        # some variants put narratives directly on the txn
        return any(k.lower().startswith("transactionnarr") for k in d.keys())

    def _walk(node: Any) -> None:
        nonlocal found
        if found:
            return
        if isinstance(node, list):
            if node and all(isinstance(x, dict) for x in node) and any(_looks_like_txn(x) for x in node):
                found = [x for x in node if isinstance(x, dict)]
                return
            for x in node:
                _walk(x)
        elif isinstance(node, dict):
            for v in node.values():
                _walk(v)

    _walk(resp)
    return found


def _find_txn_by_rrn(details: list[dict], rrn: str) -> Optional[dict]:
    """Find the transaction whose narratives reference this RRN.

    Matching is tolerant: case-insensitive, whitespace-stripped, and matches if
    the RRN equals a narrative, appears as a delimited token (/, space, |, -, :,
    comma), or is contained as a substring (for refs >= 6 chars, to avoid false
    positives on short numbers)."""
    target = _norm_ref(rrn)
    if not target:
        return None
    for txn in details:
        for nv in _narratives(txn):
            nnv = _norm_ref(nv)
            if not nnv:
                continue
            if nnv == target:
                return txn
            # delimited tokens
            tokens = re.split(r"[/\s|,:\-]+", nnv)
            if target in tokens:
                return txn
            # substring (only for sufficiently long refs)
            if len(target) >= 6 and (target in nnv or nnv in target):
                return txn
    return None


def _log_rrn_miss(p: ComplaintProcessing, details: list[dict], rrn: str) -> None:
    """When an RRN doesn't match, dump exactly what we searched vs every
    narrative seen, so a real value mismatch (e.g. 0 vs 8) is visible in logs."""
    seen = []
    for txn in details:
        seen.extend(_narratives(txn))
    log.warning("[%s] RRN miss: looking for '%s' (norm='%s'); narratives seen=%s",
                p.acknowledgement_no, rrn, _norm_ref(rrn), seen[:40])


def _rrn_validated(s: Session, p: ComplaintProcessing, incidents: list[Incident]) -> bool:
    """True only if EVERY incident's RRN was found in the DD enquiry. Used to
    decide whether a credit case may proceed to hold, or must be returned to MHA
    as 'miscellaneous'."""
    for inc in incidents:
        step = _step_get(s, p, "DD_TXN", f"DDTXN:{p.acknowledgement_no}:{inc.merchant_id}")
        if not (step and (step.response_payload or {}).get("found")):
            return False
    return True

def _dd_enquiry(s: Session, p: ComplaintProcessing, account: str) -> dict:
    """Demand-deposit / balance enquiry — returns the accounts[] structure."""
    idem = f"DD:{p.acknowledgement_no}:{account}"
    existing = _step_get(s, p, "DD_ENQUIRY", idem)
    if existing and existing.status == "SUCCESS":
        return existing.response_payload or {}
    resp = gateway().demand_deposit(account)
    _step_record(s, p, "DD_ENQUIRY", idem, {"account": account}, resp)
    return resp


def _available_balance(s: Session, p: ComplaintProcessing, account: str) -> Decimal:
    """Balance used for the hold decision. Tries the configured path first;
    if that misses, deep-searches for BALANCE_FIELD (accepting either a scalar
    or an {amount} object) and logs the real path. Raises with the response
    SHAPE (keys only) if nothing usable is found."""
    resp = _dd_enquiry(s, p, account)

    bal = _coerce_balance(_dig(resp, config.BALANCE_PATH))
    if bal is None:
        found, found_path = _deep_find_balance(resp, config.BALANCE_FIELD)
        bal = _coerce_balance(found)
        if bal is not None:
            log.warning(
                "[%s] balance not at BALANCE_PATH='%s'; found via deep search at "
                "'%s' — set BALANCE_PATH='%s' to make it explicit.",
                p.acknowledgement_no, config.BALANCE_PATH, found_path, found_path,
            )
    if bal is None:
        raise GatewayError(
            f"balance not found. tried BALANCE_PATH='{config.BALANCE_PATH}' and "
            f"deep search for '{config.BALANCE_FIELD}'. "
            f"response shape (keys only) = {_shape(resp)}"
        )
    _stash(s, p, "BALANCE", {"value": str(bal)})   # for root_effective_balance
    return bal


def _bq_locate(s: Session, p: ComplaintProcessing, account: str, inc: Incident) -> dict:
    """Find the disputed transaction in CBDOC by account + amount + date + ref."""
    idem = f"BQ:{p.acknowledgement_no}:{inc.merchant_id}"
    existing = _step_get(s, p, "BQ_TRACE", idem)
    if existing and existing.status == "SUCCESS":
        return existing.response_payload or {}
    filters = {
        config.BQ_FIELD_ACCOUNT: account,
        config.BQ_FIELD_AMOUNT:  float(inc.amount),
        config.BQ_FIELD_DATE:    int(inc.transaction_date.strftime("%Y%m%d")),
        config.BQ_FIELD_REF:     inc.merchant_id,
    }
    resp = gateway().bigquery_search(filters, limit=100)
    _step_record(s, p, "BQ_TRACE", idem, filters, resp)
    return resp


# ===========================================================================
# Hold (idempotent)
# ===========================================================================

def _apply_holds(s: Session, p: ComplaintProcessing, instrument: Instrument,
                 disputed: Decimal, balance: Decimal) -> tuple[Decimal, Decimal]:
    """Hold min(disputed, balance). Returns (amount_held, remaining_to_trace)."""
    to_hold = min(disputed, balance) if balance > 0 else Decimal("0")
    remaining = disputed - to_hold

    if to_hold <= 0:
        log.info("[%s] no balance to hold; remaining=%s", p.acknowledgement_no, remaining)
        return Decimal("0"), remaining

    account = instrument.payer_account_number
    currency = config.HOLD_CURRENCY
    amount_str = f"{to_hold:.2f}"
    # Idempotency key ties the hold to (ack, account, amount): a retry that
    # finds a SUCCESS here will NOT call apply-hold again.
    idem = f"HOLD:{p.acknowledgement_no}:{account}:{amount_str}"

    existing = _step_get(s, p, "HOLD", idem)
    if existing and existing.status == "SUCCESS":
        log.info("[%s] hold already applied (idempotent skip)", p.acknowledgement_no)
        return to_hold, remaining

    # Reserve the key first (PENDING). If another worker raced us, the UNIQUE
    # constraint makes _step_record return the existing row instead.
    step = _step_record(s, p, "HOLD", idem,
                        {"account": account, "amount": amount_str},
                        {}, status="PENDING")
    if step.status == "SUCCESS":
        return to_hold, remaining

    resp = gateway().apply_hold(
        account_number=account, currency=currency, amount=amount_str,
        hold_type=config.HOLD_TYPE, hold_till=config.HOLD_TILL_DATE,
        hold_narrative2=f"FRAUD {p.acknowledgement_no}"[:35],
    )
    step.status = "SUCCESS"
    step.response_payload = resp
    s.flush()
    log.info("[%s] hold applied amount=%s remaining=%s",
             p.acknowledgement_no, amount_str, remaining)
    return to_hold, remaining


# ===========================================================================
# Money-trail trace + collection
# ===========================================================================

def _trace_and_collect(s: Session, p: ComplaintProcessing, incidents: list[Incident],
                       remaining: Decimal) -> tuple[list[dict], bool]:
    """CBDOC reconciliation for a CREDIT fraud.

    Spec:
      1. Search CBDOC by the CREDIT RRN to locate the credit transaction.
      2. From that credit's timestamp onward, collect every DEBIT from the
         account in chronological order.
      3. Keep collecting until the outstanding `remaining` amount is reconciled.
      4. All collected debits are returned to MHA.

    Returns (collected_transactions, fully_reconciled)."""
    account = p.complaint.instrument.payer_account_number

    # 1. Pull the account's CBDOC rows (one query; sort + walk client-side).
    rows = _cbdoc_account_debits(s, p, account)
    if not rows:
        _cbdoc_probe(s, p, incidents)
        log.warning("[%s] CBDOC returned 0 rows for account=%s", p.acknowledgement_no, account)
        return [], False

    rows = sorted(rows, key=_row_dt_key)
    fraud_refs = {inc.merchant_id for inc in incidents}

    # 2. Anchor: find the CREDIT transaction by its RRN, capture its time.
    anchor_key = None
    for row in rows:
        if str(row.get(config.BQ_FIELD_REF, "")).strip() in fraud_refs:
            anchor_key = _row_dt_key(row)
            log.info("[%s] CBDOC credit RRN located at date/time=%s",
                     p.acknowledgement_no, anchor_key)
            break

    if anchor_key is None:
        # Credit RRN not in CBDOC — fall back to the MHA-reported fraud time so we
        # can still attempt reconciliation, but flag it.
        anchor_key = min((_incident_dt_key(inc) for inc in incidents), default=(0, 0))
        log.warning("[%s] credit RRN not found in CBDOC; using MHA fraud time %s as anchor",
                    p.acknowledgement_no, anchor_key)

    # 3. Collect DEBITS strictly AFTER the credit, until reconciled.
    collected: list[dict] = []
    recovered = Decimal("0")
    root = _root_ctx(s, p)
    disputed = sum((i.disputed_amount for i in incidents), Decimal("0"))
    for row in rows:
        if _row_dt_key(row) <= anchor_key:        # only after the credit
            continue
        rrn = str(row.get(config.BQ_FIELD_REF, "")).strip()
        if rrn in fraud_refs:                      # never the credit itself
            continue
        if not _is_debit(row):                     # debits only
            continue
        amt = _row_amount(row)
        if amt <= 0:
            continue
        bene = _upi_bene_for(s, p, rrn)
        collected.append(_to_mha_txn(row, root, disputed, bene=bene))
        recovered += amt
        if recovered >= remaining:
            break

    fully = recovered >= remaining
    log.info("[%s] CBDOC reconcile: total rows=%d debits-after-credit collected=%d "
             "recovered=%s remaining=%s reconciled=%s",
             p.acknowledgement_no, len(rows), len(collected), recovered, remaining, fully)
    return collected, fully


def _is_debit(row: dict) -> bool:
    """A CBDOC row is a debit (money leaving the account).
    The OG* table is outgoing payments, so every row is a debit by default.
    If your CBDOC has an explicit debit/credit indicator or signed amount,
    set BQ_DEBIT_INDICATOR_FIELD / BQ_DEBIT_INDICATOR_VALUE in config to filter."""
    field = config.BQ_DEBIT_INDICATOR_FIELD
    if not field:
        # No indicator configured: signed-amount heuristic, else treat as debit.
        raw = row.get(config.BQ_FIELD_AMOUNT)
        try:
            # if amounts are signed, negative = debit
            return Decimal(str(raw)) != 0
        except Exception:
            return True
    return str(row.get(field, "")).strip().upper() == config.BQ_DEBIT_INDICATOR_VALUE.upper()


def _cbdoc_account_debits(s: Session, p: ComplaintProcessing, account: str) -> list[dict]:
    """Fetch the account's debit transactions from CBDOC (cached per complaint).
    Filters on BQ_FIELD_ACCOUNT — set that env var to the CBDOC column that holds
    the DEBITED (originating) account. On 0 rows, runs a diagnostic probe."""
    idem = f"CBDOCDEBITS:{p.acknowledgement_no}:{account}"
    existing = _step_get(s, p, "BQ_TRACE", idem)
    if existing and existing.status == "SUCCESS":
        return _rows(existing.response_payload or {})

    filt = {config.BQ_FIELD_ACCOUNT: account}
    log.info("[%s] CBDOC account search: field BQ_FIELD_ACCOUNT=%s value=%s",
             p.acknowledgement_no, config.BQ_FIELD_ACCOUNT, account)
    resp = gateway().bigquery_search(filt, limit=config.CBDOC_MAX_ROWS)
    rows = _rows(resp)
    _step_record(s, p, "BQ_TRACE", idem, filt, resp)
    log.info("[%s] CBDOC account search returned %d rows", p.acknowledgement_no, len(rows))
    return rows


def _cbdoc_probe(s: Session, p: ComplaintProcessing, incidents: list[Incident]) -> None:
    """Diagnostic: when the account search returns nothing, locate the fraud
    transaction by its reference number and LOG that row's fields. This reveals
    how CBDOC actually stores the account so you can set BQ_FIELD_ACCOUNT (and
    the correct account value) — the single mapping this reconciliation needs."""
    for inc in incidents:
        try:
            resp = gateway().bigquery_search({config.BQ_FIELD_REF: inc.merchant_id}, limit=5)
        except Exception as e:                      # noqa: BLE001
            log.warning("[%s] CBDOC probe by %s=%s failed: %s",
                        p.acknowledgement_no, config.BQ_FIELD_REF, inc.merchant_id, e)
            continue
        rows = _rows(resp)
        log.warning("[%s] CBDOC probe %s=%s -> %d rows",
                    p.acknowledgement_no, config.BQ_FIELD_REF, inc.merchant_id, len(rows))
        if rows:
            log.warning("[%s] CBDOC fraud row fields = %s",
                        p.acknowledgement_no, {k: rows[0].get(k) for k in rows[0]})
            log.warning("[%s] ACTION: set BQ_FIELD_ACCOUNT to whichever field above holds the "
                        "account whose debits you want, and make sure the complaint's "
                        "payer_account_number matches that stored value/format.",
                        p.acknowledgement_no)
            return
    log.warning("[%s] CBDOC probe found nothing even by reference number — confirm the fraud "
                "ref (incident 'm'=%s) matches CBDOC's %s and that data exists for it.",
                p.acknowledgement_no,
                incidents[0].merchant_id if incidents else "?", config.BQ_FIELD_REF)


def _rows(bq_resp: dict) -> list[dict]:
    return bq_resp.get("rows", []) if isinstance(bq_resp, dict) else []


def _row_amount(row: dict) -> Decimal:
    raw = row.get(config.BQ_FIELD_AMOUNT)
    try:
        return abs(Decimal(str(raw))) if raw is not None else Decimal("0")
    except Exception:
        return Decimal("0")


def _row_dt_key(row: dict) -> tuple[int, int]:
    """Sort key: capture date then capture time (ints), 0 if missing."""
    def _int(v):
        try:
            return int(v)
        except Exception:
            return 0
    return (_int(row.get(config.BQ_FIELD_DATE)), _int(row.get(config.BQ_FIELD_TIME)))


def _incident_dt_key(inc: Incident) -> tuple[int, int]:
    """Fraud time from the MHA request encoded the same way as _row_dt_key:
    (YYYYMMDD, HHMMSS). Used to collect only CBDOC debits AFTER the fraud time."""
    d = inc.transaction_date
    t = inc.transaction_time
    date_i = d.year * 10000 + d.month * 100 + d.day if d else 0
    time_i = (t.hour * 10000 + t.minute * 100 + t.second) if t else 0
    return (date_i, time_i)


def _root_ctx(s: Session, p: ComplaintProcessing) -> dict:
    """The 'root' = the original fraud account + first incident's RRN + the
    account's balance. Shared across every response object."""
    inst = p.complaint.instrument
    incs = inst.incidents if inst else []
    primary = incs[0] if incs else None
    return {
        "root_account_number": inst.payer_account_number if inst else "",
        "root_rrn_transaction_id": (primary.merchant_id if primary else ""),
        "root_bankid": (inst.payer_bank_code if inst else "") or "",
        "root_effective_balance": _unstash(s, p, "BALANCE").get("value", ""),
        "root_ifsc_code": "",          # TODO: source IFSC of the root account
    }


# Fields we don't yet have a source for are emitted as "" so the JSON shape is
# complete. Wire them up once you confirm where they come from (transaction
# enquiry narratives, a customer-info service, or extra CBDOC columns).
def _normalise_bene(data: dict) -> dict:
    """Map the real UPI API response (data.*) to a canonical bene dict.
    Real UPI response keys:
      pyAccNo     -> beneficiary account
      pyIfscCode  -> beneficiary IFSC
      bankName    -> beneficiary bank (short name)
      bankCode    -> beneficiary bank code
      pyFname     -> beneficiary name
      pyFvaddr    -> beneficiary VPA
    """
    data = data or {}
    return {
        "account": data.get("pyAccNo", ""),
        "ifsc":    data.get("pyIfscCode", ""),
        "bank":    data.get("bankName", ""),
        "bankCode": data.get("bankCode", ""),
        "name":    data.get("pyFname", ""),
        "vpa":     data.get("pyFvaddr", ""),
    }


def _upi_bene_for(s: Session, p: ComplaintProcessing, rrn: str) -> dict:
    """Return normalised UPI bene data for this RRN if it was fetched, else {}."""
    step = _step_get(s, p, "UPI_BENE", f"UPI:{p.acknowledgement_no}:{rrn}")
    if step and step.status == "SUCCESS" and step.response_payload:
        return _normalise_bene(_dig(step.response_payload, "data") or {})
    return {}


def _to_mha_txn(row: dict, root: dict, disputed: Decimal,
                bene: Optional[dict] = None) -> dict:
    """A CBDOC debit leg → MHA 'Money Transfer To' object (txn_type_id 3).
    If UPI bene details were fetched (IB Debit >24h UPI path), they fill in
    payee_bank / payee_account_number / ifsc_code."""
    bene = bene or {}
    return {
        "rrn_transaction_id": row.get(config.BQ_FIELD_REF, ""),
        "payee_bank": bene.get("bank") or row.get("OGBEA1", ""),
        "payee_bank_code": bene.get("bankCode", ""),
        "payee_account_number": bene.get("account") or row.get(config.BQ_FIELD_BENE_ACCOUNT, ""),
        "payee_name": bene.get("name", ""),
        "payee_vpa": bene.get("vpa", ""),
        "amount": str(row.get(config.BQ_FIELD_AMOUNT, "")),
        "transaction_datetime": _fmt_dt(row.get(config.BQ_FIELD_DATE), row.get(config.BQ_FIELD_TIME)) or "",
        "disputed_amount": str(disputed),
        "ifsc_code": bene.get("ifsc", ""),
        "txn_type": "Money Transfer To",
        "txn_type_id": "3",
        "phone_number": "",
        "email": "",
        "pan_number": "",
        "root_account_number": root["root_account_number"],
        "root_rrn_transaction_id": root["root_rrn_transaction_id"],
        "root_bankid": root["root_bankid"],
        "status_code": "00",
        "remarks": row.get("OGNAR1", "") or "",
        "root_effective_balance": root["root_effective_balance"],
        "root_ifsc_code": root["root_ifsc_code"],
    }


def _fmt_dt(d: Any, t: Any) -> Optional[str]:
    if d is None:
        return None
    ds = str(d)
    ts = str(t).zfill(6) if t is not None else "000000"
    try:
        return (f"{ds[0:4]}-{ds[4:6]}-{ds[6:8]} "
                f"{ts[0:2]}:{ts[2:4]}:{ts[4:6]}")
    except Exception:
        return ds

def _hold_txns(s: Session, p: ComplaintProcessing, root: dict) -> list[dict]:
    """Applied holds -> MHA 'Transaction Put on Hold' objects (txn_type_id 1).
    One object per hold."""
    rows = s.execute(
        select(ProcessingStep).where(
            ProcessingStep.complaint_id == p.complaint_id,
            ProcessingStep.step == "HOLD",
            ProcessingStep.status == "SUCCESS",
        )
    ).scalars().all()
    out = []
    for r in rows:
        req = r.request_payload or {}
        out.append({
            "root_account_number": req.get("account", root["root_account_number"]),
            "root_rrn_transaction_id": root["root_rrn_transaction_id"],
            "root_bankid": root["root_bankid"],
            "ifsc_code": "",
            "amount": req.get("amount", ""),
            "transaction_datetime": "",
            "phone_number": "",
            "email": "",
            "pan_number": "",
            "remarks": "",
            "txn_type": "Transaction Put on Hold",
            "txn_type_id": "1",
            "status_code": "00",
            "root_effective_balance": root["root_effective_balance"],
            "root_ifsc_code": root["root_ifsc_code"],
        })
    return out


def _finalize(s: Session, p: ComplaintProcessing, instrument: Instrument,
              incidents: list[Incident], traced_txns: list[dict]) -> None:
    """transactions = one 'Transaction Put on Hold' per hold + one 'Money
    Transfer To' per traced CBDOC debit (already built with root context)."""
    root = _root_ctx(s, p)
    payload = {
        "acknowledgement_no": p.acknowledgement_no,
        "job_id": p.job_id,
        "transactions": _hold_txns(s, p, root) + traced_txns,
    }
    _write_response_log(s, p, payload)

def _finalize_bene_only(s: Session, p: ComplaintProcessing, incidents: list[Incident]) -> None:
    """UPI-debit: one 'Money Transfer To' per incident, populated from the UPI

    root = _root_ctx(s, p)

    txns = []
    for inc in incidents:
        step = _step_get(s, p, "UPI_BENE", f"UPI:{p.acknowledgement_no}:{inc.merchant_id}")
        raw = _dig(step.response_payload, "data") if step and step.response_payload else {}
        bene = _normalise_bene(raw or {})
        txns.append({
            "rrn_transaction_id": inc.merchant_id,
            "payee_bank": bene.get("bank", ""),
            "payee_bank_code": bene.get("bankCode", ""),
            "payee_account_number": bene.get("account", ""),
            "payee_name": bene.get("name", ""),
            "payee_vpa": bene.get("vpa", ""),

def _write_response_log(s: Session, p: ComplaintProcessing, payload: dict,
                        response_code: str = "00",
                        response_message: str = "SUCCESS") -> None:
    existing = s.execute(
        select(ApiResponseLog).where(ApiResponseLog.job_id == p.job_id)
    ).scalar_one_or_none()
    if existing:
        existing.response_payload = payload
        existing.response_code = response_code
        existing.response_message = response_message
        existing.responded_at = _now()
    else:
        s.add(ApiResponseLog(
            complaint_id=p.complaint_id, acknowledgement_no=p.acknowledgement_no,
            job_id=p.job_id, response_code=response_code, response_message=response_message,
            response_payload=payload, responded_at=_now(),
        ))
    s.flush()
    log.info("[%s] response log written job_id=%s txns=%d",
             p.acknowledgement_no, p.job_id, len(payload.get("transactions", [])))
